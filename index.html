Kodlar

wordcount.py

#!/usr/bin/env python3
"""
CSE423 Cloud Computing - Word Count with Apache Spark
Adnan Menderes University
"""

from pyspark.sql import SparkSession
import time

def create_spark_session():
    """Initialize and return Spark Session"""
    return SparkSession.builder \
        .appName("CSE423-WordCount-10GB") \
        .config("spark.sql.shuffle.partitions", "200") \
        .getOrCreate()

def perform_word_count(sc, input_path):
    """Perform MapReduce word count operation"""
    text_rdd = sc.textFile(input_path)
    
    word_counts = text_rdd \
        .flatMap(lambda line: line.lower().split()) \
        .filter(lambda word: word.isalpha() and len(word) > 1) \
        .map(lambda word: (word, 1)) \
        .reduceByKey(lambda a, b: a + b)
    
    return word_counts

def get_statistics(word_counts):
    """Calculate and return word count statistics"""
    unique_words = word_counts.count()
    total_words = word_counts.map(lambda x: x[1]).sum()
    top_100 = word_counts \
        .sortBy(lambda x: x[1], ascending=False) \
        .take(100)
    
    return unique_words, total_words, top_100

def main():
    # ⚠️ BU PATH'LERİ GÜNCELLEYİN!
    INPUT_PATH = "s3://cse423-wordcount-[OGRENCI-NO]/input/large_text_data.txt"
    OUTPUT_PATH = "s3://cse423-wordcount-[OGRENCI-NO]/output/"
    
    print("=" * 70)
    print("         CSE423 CLOUD COMPUTING - WORD COUNT PROJECT")
    print("=" * 70)
    
    spark = create_spark_session()
    sc = spark.sparkContext
    
    print(f"Spark Version: {spark.version}")
    print(f"Application ID: {sc.applicationId}")
    
    start_time = time.time()
    
    word_counts = perform_word_count(sc, INPUT_PATH)
    word_counts.cache()
    
    unique_words, total_words, top_100 = get_statistics(word_counts)
    
    end_time = time.time()
    execution_time = end_time - start_time
    
    print(f"\nTotal Unique Words: {unique_words:,}")
    print(f"Total Word Count: {total_words:,}")
    print(f"Execution Time: {execution_time:.2f} seconds")
    print(f"Processing Speed: {10 / (execution_time / 60):.2f} GB/minute")
    
    print("\nTOP 100 WORDS:")
    for i, (word, count) in enumerate(top_100, 1):
        print(f"{i:3}. {word:20} : {count:,}")
    
    # Save results
    sorted_counts = word_counts.sortBy(lambda x: x[1], ascending=False)
    sorted_counts.saveAsTextFile(OUTPUT_PATH)
    
    print(f"\nResults saved to: {OUTPUT_PATH}")
    spark.stop()

if __name__ == "__main__":
    main()

#!/usr/bin/env python3
"""
CSE423 - Synthetic Data Generator
Generates a 10GB text file for word count analysis

Usage:
    python generate_data.py         # Full 10GB
    python generate_data.py --test  # Test 100MB
"""

import random
import time
import os
import sys

COMMON_WORDS = [
    'the', 'be', 'to', 'of', 'and', 'a', 'in', 'that', 'have', 'i',
    'it', 'for', 'not', 'on', 'with', 'he', 'as', 'you', 'do', 'at',
    'this', 'but', 'his', 'by', 'from', 'they', 'we', 'say', 'her', 'she',
    'cloud', 'computing', 'data', 'server', 'cluster', 'node', 'network',
    'storage', 'memory', 'processor', 'distributed', 'parallel', 'hadoop',
    'spark', 'mapreduce', 'aws', 'azure', 'google', 'amazon', 'instance',
    'software', 'hardware', 'application', 'algorithm', 'code', 'programming',
    'and', 'or', 'but', 'so', 'because', 'if', 'when', 'where', 'which'
]

def generate_paragraph():
    num_sentences = random.randint(3, 8)
    sentences = []
    for _ in range(num_sentences):
        length = random.randint(8, 25)
        words = random.choices(COMMON_WORDS, k=length)
        sentences.append(' '.join(words))
    return '. '.join(sentences) + '.\n'

def generate_large_file(filename, target_size_gb=10):
    target_size = target_size_gb * 1024 * 1024 * 1024
    current_size = 0
    
    print(f"Generating {target_size_gb}GB file: {filename}")
    start_time = time.time()
    
    with open(filename, 'w', encoding='utf-8') as f:
        while current_size < target_size:
            paragraph = generate_paragraph()
            f.write(paragraph)
            current_size += len(paragraph.encode('utf-8'))
            
            if current_size % (1024 * 1024 * 500) < 1000:
                progress = (current_size / target_size) * 100
                elapsed = time.time() - start_time
                speed = current_size / elapsed / (1024 * 1024)
                eta = (target_size - current_size) / (current_size / elapsed) / 60
                print(f"Progress: {progress:.1f}% | Size: {current_size/1024**3:.2f}GB | Speed: {speed:.1f}MB/s | ETA: {eta:.1f}min")
    
    print(f"\nDone! File: {filename}, Size: {os.path.getsize(filename)/1024**3:.2f}GB")

if __name__ == "__main__":
    if len(sys.argv) > 1 and sys.argv[1] == '--test':
        # 100MB test file
        target = 100 * 1024 * 1024
        current = 0
        with open("test_data.txt", 'w') as f:
            while current < target:
                p = generate_paragraph()
                f.write(p)
                current += len(p.encode('utf-8'))
        print(f"Test file created: test_data.txt ({os.path.getsize('test_data.txt')/1024**2:.1f}MB)")
    else:
        generate_large_file("large_text_data.txt", 10)

Çözüm: Spark ayarlarını değiştir

spark-submit \
    --executor-memory 6g \
    --driver-memory 4g \
    --conf spark.sql.shuffle.partitions=400 \
    wordcount.py

S3 Access Denied hatası

Çözümler:

  • • Region'un us-east-1 olduğundan emin ol
  • • AWS credentials'ın güncel olduğunu kontrol et
  • • Learner Lab session'ın aktif olduğundan emin ol

Bütçe yetersiz

Tasarruf yöntemleri:

  • • Daha küçük veri kullan (5GB)
  • • Daha az node kullan (2 node)
  • • m4.large instance kullan
  • • Cluster'ı hemen terminate et
